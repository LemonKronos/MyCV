    (0,0) -- (2,0) -- (2.2,0.2) -- (2,0.4) -- (0,0.4) -- cycle;

    (0,0) -- (4,0) -- (4.2,0.2) -- (4,0.4) -- (0,0.4) -- (0.2, 0.2) -- cycle;
    \node[text=white, anchor=center] at (2,0.18) {#2};
  \end{tikzpicture}%

    \node[text=white, anchor=center] at (#4 / 2 + 0.1,0.18) {#3};
  \end{tikzpicture}%
}
