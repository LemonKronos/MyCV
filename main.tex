\newcommand{\powerprompt}[4]{%

    \filldraw[fill=yellow, draw=none]
    (0,0) -- (2,0) -- (2.2,0.2) -- (2,0.4) -- (0,0.4) -- cycle;
    \node[text=black, anchor=center] at (1,0.18) {#1};
  \end{tikzpicture}%
  \begin{tikzpicture}[baseline]
    \filldraw[fill=blue, draw=none]
    (0,0) -- (4,0) -- (4.2,0.2) -- (4,0.4) -- (0,0.4) -- (0.2, 0.2) -- cycle;
    \node[text=white, anchor=center] at (2,0.18) {#2};
  \end{tikzpicture}%
  \begin{tikzpicture}[baseline]
    \filldraw[fill=cyan, draw=none]
    (0,0) -- (#4,0) -- (#4 + 0.2,0.2) -- (#4,0.4) -- (0,0.4) -- (0.2,0.2) -- cycle;
    \node[text=white, anchor=center] at (#4 / 2 + 0.1,0.18) {#3};
  \end{tikzpicture}%
}
